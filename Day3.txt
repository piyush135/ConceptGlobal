db.getCollection("customers").aggregate(

	// Pipeline
	[
		// Stage 1
		{
			$group: {
				_id: "$country",
				NoofC: {$sum:1 }
				
			}
		},

		// Stage 2
		{
			$group: {
				_id: "$NoofC",
				noofcountries: {$sum :1}
			}
		},

	]

	// Created with Studio 3T, the IDE for MongoDB - https://studio3t.com/

);
